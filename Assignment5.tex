    \end{block}

\section{Solving}
\begin{frame}
\begin{enumerate}
\item Finding $E(X)$.{
\begin{align}
E(X)&=\sum XP(X) \\
&=0\times P(0)+1\times P(1) \\
&=P(1)\\
&=0.7
\end{align}
}
\end{enumerate}
 
\end{frame}
\begin{frame}
\begin{enumerate}
\item Finding $Var(X)$.{
\begin{align}
Var(X)&=E(X^2)-(E(X))^2\\
&=\sum X^2P(X^2)-(0.7)^2\\
&=0^2\times P(0^2)+1^2\times P(1^2)-0.49\\
&=0+(0.7)-0.49\\
&=0.21
\end{align}
}
\end{enumerate}

$\therefore$ E(X) = 0.7 and Var(X) = 0.21
\end{frame}
